<script>
const API_URL = "https://script.google.com/macros/s/AKfycbygia2T_Hbh9H9AkCK_IEykgwSSxBoGysdj-y6BsDHaT2PKYu5iBRx14Ea1ign8MU23/exec";

const status = document.getElementById("status");
const busquedaInput = document.getElementById("busqueda");
const buscarBtn = document.getElementById("buscarBtn");
const toggleFiltrosBtn = document.getElementById("toggleFiltrosBtn");
const filtrosPanel = document.getElementById("filtrosPanel");
const filtroColor = document.getElementById("filtroColor");
const filtroBateria = document.getElementById("filtroBateria");
const aplicarFiltrosBtn = document.getElementById("aplicarFiltrosBtn");
const limpiarFiltrosBtn = document.getElementById("limpiarFiltrosBtn");
const backTopBtn = document.getElementById("backTopBtn");

let todosLosEquipos = [];

function jsonp(params){
  return new Promise((resolve,reject) => {
    const callback = "__nicphones_" + Date.now() + "_" + Math.random().toString(36).slice(2);
    const script = document.createElement("script");
    const query = new URLSearchParams({ ...params, callback, t:Date.now() });
    const timeout = setTimeout(() => {
      cleanup();
      reject(new Error("No se pudo cargar la informacion"));
    }, 15000);

    function cleanup(){
      clearTimeout(timeout);
      delete window[callback];
      script.remove();
    }

    window[callback] = data => {
      cleanup();
      resolve(data);
    };

    script.onerror = () => {
      cleanup();
      reject(new Error("Error de conexion"));
    };

    script.src = API_URL + "?" + query.toString();
    document.body.appendChild(script);
  });
}

function normalizarTexto(texto){
  return (texto || "").toString().trim().toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g,"");
}

function modeloCorto(modelo){
  return (modelo || "").toString().replace(/^\s*iphone\s+/i,"").trim();
}

function claseColor(color){
  const c = normalizarTexto(color);
  if(c.includes("blue titanium")) return "color-titanium-blue";
  if(c.includes("black titanium")) return "color-titanium-black";
  if(c.includes("white titanium")) return "color-titanium-white";
  if(c.includes("natural")) return "color-natural";
  if(c.includes("midnight") || c.includes("black") || c.includes("graphite")) return "color-dark";
  if(c.includes("blue")) return "color-blue";
  if(c.includes("pink")) return "color-pink";
  if(c.includes("white") || c.includes("starlight")) return "color-white";
  if(c.includes("silver")) return "color-silver";
  if(c.includes("gold")) return "color-gold";
  if(c.includes("green")) return "color-green-pill";
  if(c.includes("purple")) return "color-purple";
  if(c.includes("red")) return "color-red-pill";
  return "color-default";
}

function renderTabla(equipos){
  const tbody = document.getElementById("tabla");
  tbody.innerHTML = "";

  equipos.forEach(eq => {
    tbody.innerHTML += `
      <tr data-search="${eq.search}">
        <td data-label="MODELO"><strong>${modeloCorto(eq.modelo)}</strong></td>
        <td data-label="COLOR"><span class="color-pill ${claseColor(eq.color)}">${eq.color}</span></td>
        <td data-label="GB"><span class="badge">${eq.gb}</span></td>
        <td data-label="BATERIA" class="${eq.claseBateria}">${eq.bateria}%</td>
        <td data-label="USD" class="final-usd">$${eq.usdFinal}</td>
        <td data-label="PESOS" class="price-ars">$${Number(eq.pesos).toLocaleString()}</td>
      </tr>`;
  });
}

function poblarFiltroColor(equipos){
  const colorActual = filtroColor.value;
  const colores = [...new Set(equipos.map(eq => (eq.color || "").trim()).filter(Boolean))]
    .sort((a,b) => a.localeCompare(b));

  filtroColor.innerHTML = '<option value="">Todos</option>';
  colores.forEach(color => {
    const option = document.createElement("option");
    option.value = color;
    option.textContent = color;
    filtroColor.appendChild(option);
  });
  if(colores.includes(colorActual)) filtroColor.value = colorActual;
}

function aplicarFiltros(){
  const colorElegido = normalizarTexto(filtroColor.value);
  const bateriaElegida = filtroBateria.value;

  const filtrados = todosLosEquipos.filter(eq => {
    const coincideColor = !colorElegido || normalizarTexto(eq.color) === colorElegido;
    let coincideBateria = true;
    if(bateriaElegida === "100") coincideBateria = eq.bateria === 100;
    else if(bateriaElegida) coincideBateria = eq.bateria > parseInt(bateriaElegida);
    return coincideColor && coincideBateria;
  });

  renderTabla(filtrados);
  const hayFiltros = Boolean(colorElegido || bateriaElegida);
  status.textContent = hayFiltros
    ? `Filtrado: ${filtrados.length} de ${todosLosEquipos.length} equipos`
    : `Stock cargado: ${todosLosEquipos.length} equipos`;
}

async function cargar(){
  try{
    status.textContent = "Cargando stock...";
    const data = await jsonp({ mode:"public" });
    if(!data.ok) throw new Error(data.error || "No se pudo cargar stock");
    todosLosEquipos = data.equipos || [];
    poblarFiltroColor(todosLosEquipos);
    aplicarFiltros();
  }catch(e){
    console.log(e);
    status.textContent = "Error al cargar stock";
  }
}

function buscarFila(){
  const texto = normalizarTexto(busquedaInput.value);
  if(!texto) return;
  const filas = document.querySelectorAll("[data-search]");
  filas.forEach(fila => fila.style.outline = "none");

  for(const fila of filas){
    if((fila.dataset.search || "").includes(texto)){
      fila.scrollIntoView({ behavior:"smooth", block:"center" });
      fila.style.outline = "3px solid #00ff88";
      return;
    }
  }
  status.textContent = "No se encontro ningun resultado";
}

buscarBtn.addEventListener("click",buscarFila);
toggleFiltrosBtn.addEventListener("click",() => filtrosPanel.classList.toggle("open"));
aplicarFiltrosBtn.addEventListener("click",aplicarFiltros);
limpiarFiltrosBtn.addEventListener("click",() => {
  filtroColor.value = "";
  filtroBateria.value = "";
  aplicarFiltros();
});
busquedaInput.addEventListener("keydown",e => { if(e.key === "Enter") buscarFila(); });
backTopBtn.addEventListener("click",() => window.scrollTo({ top:0, behavior:"smooth" }));
cargar();
